# useClickOutside

Detects click and touch events outside of given element

<br />

<br />

#### Demo

<DemoUseClickOutside />

```tsx page.tsx
import { useState } from 'react'
import { useClickOutside } from '@codecentrum/piksel-hooks'
import { Button } from '@codecentrum/piksel-ui'

export default function DemoUseClickOutside() {
  const [opened, setOpened] = useState('')
  const ref = useClickOutside(() => setOpened('Click Outside'))

  return (
    <>
      <Button themes="info" ref={ref} onClick={() => setOpened('Click Inside')}>
        Button
      </Button>
      <p>{opened}</p>
    </>
  )
}
```

<br />

#### Definition

```tsx
function useClickOutside<T extends HTMLElement = any>(
  handler: () => void,
  events?: string[] | null,
  nodes?: (HTMLElement | null)[]
): import('react').MutableRefObject<T | undefined>
```
